#!/bin/bash

apt update
apt dist-upgrade

'sudo -i' >  /etc/motd/newOSversion.sh
'do-release-upgrade -d' >>  /etc/motd/newOSversion.sh
'rm -- "$0"' >>  /etc/motd/newOSversion.sh

"'apt-get update' /etc/motd/justTemp.sh" >>  /etc/motd/newOSversion.sh
"'apt-get install upgrade' /etc/motd/justTemp.sh" >>  /etc/motd/newOSversion.sh
"'rm -rf /etc/motd/newOSversion.sh' /etc/motd/justTemp.sh" >>  /etc/motd/newOSversion.sh
"'rm -- ´$0´' /etc/motd/justTemp.sh" >>  /etc/motd/newOSversion.sh
reboot
